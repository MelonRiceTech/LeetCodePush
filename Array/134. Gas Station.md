####134. Gas Station

##### Difficulty: Medium

```
class Solution {
    public int canCompleteCircuit(int[] gas, int[] cost) {
        if (gas.length == 0){
            return -1;
        }
        int sum = 0;
        for (int i = 0;i < gas.length;i++){
            sum += gas[i];
            sum -= cost[i];
        }
        if (sum < 0){
            return -1;
        }else{
            sum = 0;
            int res = 0;
            for (int i = 0;i < gas.length;i++){
                sum += gas[i] - cost[i];
                if (sum < 0){
                    sum = 0;
                    res = (i + 1)% gas.length;
                }
            }
            return res;
        }
    }
}
```
